int convertkey(char *id){
    int sum =0;
    for(int i=0;i< strlen(id);i++){
       
       int num =(int)id[i]*(i+1);
       sum= sum+num;
    }
  
   return sum%2069;
}
int search(int key) {

	while(hashArray[hashIndex] != NULL) {

}
